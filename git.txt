sudo apt-get install git-core

기본 설정 리스트
git config --list

사용자 정보 추가
git config --global user.name "makuvex" 
git config --global user.email "makuvex7@gmail.com"

컬러 설정
git config --global color.ui "auto"

git init
git add * 
git status

git commit -m "first"

git log

push
